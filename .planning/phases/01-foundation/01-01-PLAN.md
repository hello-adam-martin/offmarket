```typescript
"use client";

import { ThemeProvider as NextThemesProvider } from "next-themes";

export function ThemeProvider({ children }: { children: React.ReactNode }) {
  return (
    <NextThemesProvider
      attribute="class"
      defaultTheme="system"
      enableSystem
      disableTransitionOnChange
    >
      {children}
    </NextThemesProvider>
  );
}
```

Settings per UI-SPEC.md:
- `attribute="class"` — adds/removes `.dark` class on `<html>` (matches Tailwind `darkMode: 'selector'`)
- `defaultTheme="system"` — respect OS preference on first visit
- `enableSystem` — three-way toggle: Light / Dark / System
- `disableTransitionOnChange` — prevent CSS transitions firing during theme switch
